Note:_____________________________________ # rrespino.com 2017 # Cleanup for iframe rendering # #--------------------------------------------------------------------------------- #/////////////////////////////////////////////////////////// from urllib.request import urlopen from bs4 import BeautifulSoup jobType = input("Please enter job Type\n") city = input("Please enter job city\n") state = input("Please enter state initals\n") URL = "https://www.indeed.com/jobs?q="+jobType+"&l="+city+"%2C"+state print ('URL', URL) soup = BeautifulSoup(urlopen(URL).read(), 'html.parser') results = soup.find_all('div', attrs={'data-tn-component': 'organicJob'}) #for all sets returned in result from web scraper for x in results: company = x.find('span', attrs={"itemprop":"name"}) print ('company:', company.text.strip()) job = x.find('a', attrs={'data-tn-element': "jobTitle"}) print ('job:', job.text.strip()) salary = x.find('nobr') if salary: print ('salary:', salary.text.strip()) print ('----------') #example output for parameters (software engineer, boulder, co) #
